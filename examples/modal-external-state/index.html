Modal External State

Dependencies

Type Name URL
scriptmithril@2.0.4https://unpkg.com/mithril@2.0.4

JS #

let state = {
  modal: null
}

const Modal = {
  view: () =>
    m('div.modal',
      m('div.c1',
        m('div.c2', state.modal.text),
        m('button', {onclick: () => state.modal = null}, 'OK')
      )
    )
}

const App = {
  view: () => [
    m('h1', 'TEST'),
    m('button', {onclick: () => state.modal = {text: 'This is a modal'}}, 'show modal'),
    state.modal && m(Modal)
  ]
}

m.mount(document.body, App)
