~~~js
class app {
  constructor() {
    this.count = 0
  }

  view() {
    return [
      m('div',
        m('h1', `Count: ${this.count}`),
        m('button', {onclick: () => this.count++}, '+'),
        m('button', {onclick: () => this.count--}, "-")
      )
    ]
  }
}

m.mount(document.body, app)
~~~
