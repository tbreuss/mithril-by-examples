~~~js
const model = {
  data: []
}

const Loading = {view: () => m('h1', "Loading...")}

const Main = {
  oninit: () => {
    // fetch data here
  },
  view: () => [
    !model.data.length
      ? m(Loading)
      : m('div', "The normal page")
  ]
}

m.mount(document.body, Main)
~~~
