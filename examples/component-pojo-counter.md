~~~js
const pojo = {
  oninit: v => {v.state.count = 0},
  view: v =>
    m('div',
      m('h1', `Count: ${v.state.count}`),
      m('button', {onclick: () => v.state.count++}, "+"),
      m('button', {onclick: () => v.state.count--}, "-")
    )
}

m.mount(document.body, pojo)
~~~
