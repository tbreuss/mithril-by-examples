In this example we can see an example of Mithril's m.mount API method, besides it core m() hyperscript function.

Ready? Here we go!

Simple Closure Component

Dependencies

Type Name URL

JavaScript #

const closure = () => {
  let count = 0

  return {
    view: () => [
      m('div',
        m('h1', `Count: ${count}`),
        m('button', {onclick: () => count++}, "+"),
        m('button', {onclick: () => count--}, "-")
      )
    ]
  }
}

m.mount(document.body, closure)
