~~~js
const app = () => {
  let text = ''

  return {
    view: () => [
      m('h1', `text: ${text}`),
      m('input', {
        oninput: e => text = e.target.value,
        value: text
      })
    ]
  }
}

m.mount(document.body, app)
~~~
